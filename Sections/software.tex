	\cvitem{\href{https://github.com/spack/spack}{spack}}{
	  De-facto standard for HPC package management. 1,400+ contributors, 8,100+ packages.\newline
          Thousands of users including HPC Centers, research labs, computational scientists,
          developers.
        }
	\cvitem{\href{https://github.com/archspec/archspec}{archspec}}{
	  Tool for assessing and reasoning about compatibility of microarchitectures.
          Used by Spack, then later Conda and Pixi package managers.
        }
	\cvitem{\href{https://github.com/spack/sbang}{sbang}}{
                Tool that allows scripts to run with very long shebang lines.
        }
	\cvitem{\github{cram}}{
		Tool for running ensembles of millions of small MPI jobs for
		Uncertainty Quantification.}
	\cvitem{\github{rubik}}{
		Tool for building optimized task mappings for structured simulation codes on supercomputers.}
	\cvitem{\github{callpath}}{
		Library for analyzing stack traces scalably in distributed applications.}
	\cvitem{\github{wrap}}{
		Code generator for MPI profiling wrapper libraries.  Used in Allinea DDT.	}
	\cvitem{\github{perf-dump}}{
		Dump per-process, per-thread PAPI counter data from MPI programs to HDF5 files.}
	\cvitem{\github{muster}}{
		Scalable $O(log(P))$ parallel clustering library, used in performance/debugging tools.}
	\cvitem{\github{nami}}{
		Parallel and sequential wavelet compression library.}
	\cvitem{\href{http://github.com/tgamblin/libra}{libra}}{
		Scalable load-balance measurement/viewer tool for MPI simulations.}
